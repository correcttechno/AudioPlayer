#include <driver/i2s.h>
#include <Arduino.h>
#define I2S_WS   26   // LRCLK
#define I2S_SD   32   // DATA
#define I2S_SCK  25   // BCLK

HardwareSerial HC12(2);   // GPIO16 RX, GPIO17 TX

const int SAMPLE_RATE = 8000;

void setup() {
  Serial.begin(115200);
  HC12.begin(115200, SERIAL_8N1, 16, 17);

  Serial.println("\n=== INMP441 I2S Test - Channel & 32-bit Versiyon ===");

  i2s_config_t i2s_config = {
    .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
    .sample_rate = SAMPLE_RATE,
    .bits_per_sample = I2S_BITS_PER_SAMPLE_32BIT,     // ← 32-bit dene
    .channel_format = I2S_CHANNEL_FMT_ONLY_RIGHT,     // ← Önce RIGHT dene
    .communication_format = I2S_COMM_FORMAT_I2S,
    .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
    .dma_buf_count = 8,
    .dma_buf_len = 64,
    .use_apll = false
  };

  i2s_pin_config_t pin_config = {
    .bck_io_num = I2S_SCK,
    .ws_io_num = I2S_WS,
    .data_out_num = I2S_PIN_NO_CHANGE,
    .data_in_num = I2S_SD
  };

  i2s_driver_install(I2S_NUM_0, &i2s_config, 0, NULL);
  i2s_set_pin(I2S_NUM_0, &pin_config);

  Serial.println("I2S başlatıldı.");
  Serial.println("Lütfen mikrofona konuş, el çırp veya üfle...");
}

void loop() {
  int32_t samples[64];
  size_t bytesRead = 0;

  i2s_read(I2S_NUM_0, samples, sizeof(samples), &bytesRead, 100);

  if (bytesRead > 0) {
    int numSamples = bytesRead / 4;   // 32-bit = 4 byte

    uint8_t sendBuf[64];
    bool soundDetected = false;
    int32_t maxLevel = 0;

    for (int i = 0; i < numSamples; i++) {
      // 32-bit → 8-bit çevir (basit)
      int32_t sample = samples[i] >> 16;          // yüksek 16 biti al
      uint8_t audioByte = (uint8_t)((sample >> 8) + 128);

      sendBuf[i] = audioByte;

      if (abs(sample) > 500) soundDetected = true;
      if (abs(sample) > maxLevel) maxLevel = abs(sample);
    }

    // HC-12 ile gönder (4'lü paket yerine şimdilik direkt)
    HC12.write(sendBuf, numSamples);

    // Ses seviyesi takibi
    if (soundDetected) {
      Serial.printf("SES ALGILANDI! Seviye: %d\n", maxLevel);
    } else if (maxLevel > 100) {
      Serial.printf("Zayıf sinyal: %d\n", maxLevel);
    }
  }
}
